with open("./train_polyline.csv") as read_file,
    open("./train_aug_polyline_raw.csv", 'w') as write_file

    for line in read_file:
        n_poly = line.count('],[') + 1
        number_of_samples = int(math.ceil(math.log(n_poly / 5 + 1) + 1))

        for sample in range(0, number_of_samples):
            if n_poly == 1:
                num_poly_samp = 1
            else:
                num_poly_samp = int(math.ceil(random.uniform(1, n_poly - 1)))
            new_poly = line.split('],[')[0:num_poly_samp]
            write_file.write("\"" + "\",\"".join(new_poly) + "\"\n")

{

         second = clust[2],
         last = clust[length(clust)])
